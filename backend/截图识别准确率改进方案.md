# 截图识别准确率改进方案

## 当前实现分析

### 现有识别流程
1. **策略选择**：自动选择单阶段或两阶段识别
2. **两阶段识别**：
   - 阶段1：检测所有题目及其边界
   - 阶段2：对每个题目进行识别（使用完整图片，不裁剪）
3. **单阶段识别**：直接识别整张图片
4. **降级策略**：如果主策略失败，尝试备用策略
5. **OCR兜底**：如果所有视觉识别失败，使用Tesseract OCR + 文本LLM

### 当前存在的问题
根据用户反馈"题目截图还是不准"，可能的原因：
1. **截图质量问题**：模糊、倾斜、光线不足
2. **截图范围问题**：不完整、包含多题、包含无关内容
3. **图像预处理不足**：没有去噪、增强对比度、矫正倾斜
4. **识别提示词不够精确**：LLM可能理解不准确
5. **缺少用户反馈机制**：无法让用户手动调整识别结果

## 改进方案

### 方案1：增强图像预处理（推荐优先实施）

#### 1.1 添加图像质量检测
```python
def analyze_image_quality(image_bytes: bytes) -> dict:
    """分析图像质量，给出改进建议"""
    img = Image.open(io.BytesIO(image_bytes))
    
    # 转换为灰度图进行分析
    gray = img.convert('L')
    np_img = np.array(gray)
    
    # 检测模糊度（使用拉普拉斯方差）
    laplacian_var = cv2.Laplacian(np_img, cv2.CV_64F).var()
    is_blurry = laplacian_var < 100
    
    # 检测亮度
    mean_brightness = np_img.mean()
    is_too_dark = mean_brightness < 80
    is_too_bright = mean_brightness > 200
    
    # 检测对比度
    contrast = np_img.std()
    is_low_contrast = contrast < 40
    
    # 检测倾斜（使用霍夫变换检测直线）
    edges = cv2.Canny(np_img, 50, 150)
    lines = cv2.HoughLines(edges, 1, np.pi/180, 100)
    angles = []
    if lines is not None:
        for line in lines[:10]:  # 只检查前10条线
            rho, theta = line[0]
            angle = np.degrees(theta) - 90
            if abs(angle) < 45:  # 只考虑接近水平的线
                angles.append(angle)
    
    avg_angle = np.mean(angles) if angles else 0
    is_skewed = abs(avg_angle) > 2  # 倾斜超过2度
    
    return {
        "is_blurry": is_blurry,
        "blur_score": float(laplacian_var),
        "is_too_dark": is_too_dark,
        "is_too_bright": is_too_bright,
        "brightness": float(mean_brightness),
        "is_low_contrast": is_low_contrast,
        "contrast": float(contrast),
        "is_skewed": is_skewed,
        "skew_angle": float(avg_angle),
        "warnings": []
    }
```

#### 1.2 自动图像增强
```python
def enhance_image(image_bytes: bytes, quality_info: dict) -> bytes:
    """根据质量分析结果自动增强图像"""
    img = Image.open(io.BytesIO(image_bytes))
    np_img = np.array(img)
    
    # 如果是彩色图，转换为灰度图（数学题通常不需要颜色）
    if len(np_img.shape) == 3:
        np_img = cv2.cvtColor(np_img, cv2.COLOR_RGB2GRAY)
    
    # 矫正倾斜
    if quality_info["is_skewed"]:
        angle = quality_info["skew_angle"]
        logger.info(f"Correcting skew: {angle:.2f} degrees")
        (h, w) = np_img.shape[:2]
        center = (w // 2, h // 2)
        M = cv2.getRotationMatrix2D(center, angle, 1.0)
        np_img = cv2.warpAffine(np_img, M, (w, h), 
                                flags=cv2.INTER_CUBIC,
                                borderMode=cv2.BORDER_REPLICATE)
    
    # 增强对比度
    if quality_info["is_low_contrast"]:
        logger.info("Enhancing contrast using CLAHE")
        clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8,8))
        np_img = clahe.apply(np_img)
    
    # 调整亮度
    if quality_info["is_too_dark"]:
        logger.info("Brightening dark image")
        np_img = cv2.convertScaleAbs(np_img, alpha=1.3, beta=30)
    elif quality_info["is_too_bright"]:
        logger.info("Darkening bright image")
        np_img = cv2.convertScaleAbs(np_img, alpha=0.8, beta=-20)
    
    # 去噪（如果图像模糊）
    if quality_info["is_blurry"]:
        logger.info("Applying denoising")
        np_img = cv2.fastNlMeansDenoising(np_img, h=10)
    
    # 锐化
    kernel = np.array([[-1,-1,-1],
                       [-1, 9,-1],
                       [-1,-1,-1]])
    np_img = cv2.filter2D(np_img, -1, kernel)
    
    # 二值化（可选，对于某些低质量图片有帮助）
    # _, np_img = cv2.threshold(np_img, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    
    # 转回PIL Image
    enhanced_img = Image.fromarray(np_img)
    
    # 保存为JPEG
    buf = io.BytesIO()
    enhanced_img.save(buf, format="JPEG", quality=95)
    return buf.getvalue()
```

### 方案4：智能重试机制

```python
async def recognize_with_retry(
    image_bytes: bytes,
    filename: str = "",
    max_retries: int = 2
) -> dict:
    """带重试的识别，每次重试使用不同的增强策略"""
    
    strategies = [
        {"auto_enhance": True, "strategy": RecognitionStrategy.AUTO},
        {"auto_enhance": True, "strategy": RecognitionStrategy.TWO_STAGE},
        {"auto_enhance": False, "strategy": RecognitionStrategy.SINGLE_STAGE},
    ]
    
    best_result = None
    best_confidence = 0
    
    for i, strategy in enumerate(strategies[:max_retries + 1]):
        try:
            logger.info(f"Recognition attempt {i+1}/{max_retries+1} with strategy: {strategy}")
            result = await recognize_questions(image_bytes, filename, **strategy)
            
            if not result.get("questions"):
                continue
            
            # 计算平均置信度
            avg_confidence = sum(q.get("confidence", 0) for q in result["questions"]) / len(result["questions"])
            
            if avg_confidence > best_confidence:
                best_result = result
                best_confidence = avg_confidence
            
            # 如果置信度足够高，直接返回
            if avg_confidence >= 0.85:
                logger.info(f"High confidence result achieved: {avg_confidence:.2f}")
                return best_result
                
        except Exception as e:
            logger.warning(f"Attempt {i+1} failed: {e}")
            continue
    
    if best_result:
        logger.info(f"Returning best result with confidence: {best_confidence:.2f}")
        return best_result
    else:
        raise ValueError("所有识别尝试均失败")
```
